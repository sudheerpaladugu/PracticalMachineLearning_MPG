pml_write_files = function(x) {
    n = length(x)
    for (i in 1:n) {
         #file name
        filename = paste0("problem_id_", i, ".txt")
        write.table(x[i], file = filename, quote = FALSE, row.names = FALSE, 
            col.names = FALSE)
    }
}
